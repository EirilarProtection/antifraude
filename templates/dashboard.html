document.getElementById("totalFrauds").innerText = formatNumber(data.total_frauds || 0); document.getElementById("avgRisk").innerText = data.avg_risk || 0; updateCharts(data); }catch(err){ console.error(err); } } // ===================================== // ORDERS // ===================================== async function loadOrders(){ try{ const response = await fetch("/api/orders"); const orders = await response.json(); const tbody = document.getElementById("ordersTable"); tbody.innerHTML = ""; orders.forEach(order=>{ let riskClass = "badge-success"; if(order.risk_score >= 70) riskClass = "badge-danger"; else if(order.risk_score >= 40) riskClass = "badge-warning"; tbody.innerHTML += ` #${order.id} ${order.client_name} R$ ${order.amount} ${order.country} ${order.risk_score} ${order.status} `; }); }catch(err){ console.error(err); } } // ===================================== // ALERTAS // ===================================== async function loadAlerts(){ try{ const response = await fetch("/api/alerts"); const alerts = await response.json(); const container = document.getElementById("alertsContainer"); container.innerHTML = ""; alerts.forEach(alert=>{ let css = "alert-low"; if(alert.severity === "HIGH") css = "alert-high"; if(alert.severity === "MEDIUM") css = "alert-medium"; container.innerHTML += `
${alert.title}
`; }); }catch(err){ console.error(err); } } // ===================================== // CHARTS // ===================================== function createCharts(){ const ctx1 = document.getElementById("transactionsChart"); transactionsChart = new Chart(ctx1,{ type:"line", data:{ labels:[], datasets:[ { label:"Transações", data:[], borderWidth:3, tension:.4 }, { label:"Fraudes", data:[], borderWidth:3, tension:.4 }] }, options:{ responsive:true } }); const ctx2 = document.getElementById("riskChart"); riskChart = new Chart(ctx2,{ type:"doughnut", data:{ labels:[ "Baixo", "Médio", "Alto" ], datasets:[ { data:[0,0,0] }] } }); } // ===================================== // UPDATE CHARTS // ===================================== function updateCharts(data){ if(data.daily_labels){ transactionsChart.data.labels = data.daily_labels; transactionsChart.data.datasets[0].data = data.daily_orders; transactionsChart.data.datasets[1].data = data.daily_frauds; transactionsChart.update(); } if(data.risk_distribution){ riskChart.data.datasets[0].data = [ data.risk_distribution.low, data.risk_distribution.medium, data.risk_distribution.high ]; riskChart.update(); } } // ===================================== // FORMAT // ===================================== function formatNumber(number){ return new Intl.NumberFormat( 'pt-BR' ).format(number); }
